Emails

/v1/email/id/get

Retrieve an Email by id and return it's details.

'id' => $this->id,
'xid' => $this->xid,
'state' => $this->state,
'date' => $this->date,
'client' => $this->client_id ? $this->client->get_data($level+1) : NULL,
'user' => $this->from_user_id ? $this->from_user->get_data($level+1) : NULL,
'from' => $this->from,
'to' => $this->to,
'subject' => $this->subject
